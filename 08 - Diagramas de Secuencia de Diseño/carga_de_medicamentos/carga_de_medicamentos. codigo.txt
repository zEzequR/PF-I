@startuml
actor Administrador
participant NegocioControlador as Controlador
participant Medicamento

== Inicia el caso de uso ==
Administrador -> Controlador : cargarNuevoMedicamento(codBarra,\n nombre, descripcion,\n laboratorio, precioUnit, stock)

== Validación de datos ==
Controlador -> Medicamento : validarDatos(codBarra,\n nombre, precioUnit, stock)

alt Datos inválidos
    Medicamento --> Controlador : errorValidacion
    Controlador --> Administrador : notificarError("Datos inválidos")
    return
end

== Creación del objeto ==
Controlador -> Medicamento : setMedicamento(codBarra,\n nombre, descripcion,\n laboratorio, precioUnit, stock)

== Registro del medicamento ==
Controlador -> Medicamento : registrar()

alt Error al registrar
    Medicamento --> Controlador : errorRegistro
    Controlador --> Administrador : notificarError("Error al registrar medicamento")
    return
end

== Confirmación ==
Medicamento --> Controlador : registroExitoso
Controlador --> Administrador : confirmar("Medicamento creado con éxito")

@enduml
