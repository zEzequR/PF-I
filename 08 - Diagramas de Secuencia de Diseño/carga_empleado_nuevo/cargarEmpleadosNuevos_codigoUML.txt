@startuml
actor Administrador
participant NegocioControlador as Controlador
participant Usuario
participant Empleado

== Inicio ==
Administrador -> Controlador : cargarNuevoEmpleado(nombreUsuario,\n contraseña, tipoUsuario)

== Validación de datos ==
Controlador -> Usuario : validarDatos(nombreUsuario, contraseña, tipoUsuario)

alt Datos inválidos
    Usuario --> Controlador : errorValidacion
    Controlador --> Administrador : notificarError("Datos inválidos")
    return
end

== Verificar usuario existente ==
Controlador -> Usuario : existeUsuario(nombreUsuario)

alt Usuario ya existente
    Usuario --> Controlador : usuarioDuplicado
    Controlador --> Administrador : notificarError("Empleado ya registrado")
    return
end

== Crear usuario ==
Controlador -> Usuario : crear(nombreUsuario, contraseña, tipoUsuario)

alt Error al crear usuario
    Usuario --> Controlador : errorCreacion
    Controlador --> Administrador : notificarError("Error al registrar usuario")
    return
end

== Crear empleado ==
Controlador -> Empleado : crearEmpleado(nombreUsuario, tipoUsuario)

alt Error al crear empleado
    Empleado --> Controlador : errorCreacion
    Controlador --> Administrador : notificarError("Error al registrar empleado")
    return
end

== Confirmación ==
Empleado --> Controlador : empleadoRegistrado
Controlador --> Administrador : confirmar("Empleado creado con éxito")

@enduml
